Step 1: Replace the current sanitizer with a stronger one

Add this exact helper function near the PDF export code:

def sanitize_pdf_text(text):
    if text is None:
        return ""
    text = str(text)

    replacements = {
        "\u2022": "-",
        "\u2013": "-",
        "\u2014": "-",
        "\u2015": "-",
        "\u2212": "-",
        "\u2018": "'",
        "\u2019": "'",
        "\u201c": '"',
        "\u201d": '"',
        "\u2026": "...",
        "\u2192": "->",
        "\u2190": "<-",
        "\u2264": "<=",
        "\u2265": ">=",
        "\u00a0": " ",
        "\u00ae": "(R)",
        "\u2122": "(TM)",
    }

    for old, new in replacements.items():
        text = text.replace(old, new)

    # Final safety pass for Helvetica/core PDF fonts.
    # This removes or replaces any remaining unsupported characters.
    text = text.encode("latin-1", "replace").decode("latin-1")
    text = text.replace("?", "-")

    return text

def pdf_cell_safe(pdf, w, h, txt="", border=0, ln=0, align="", fill=False, link=""):

def pdf_multi_cell_safe(pdf, w, h, txt="", border=0, align="", fill=False):
